286  template<typename Rng>
287  auto unzip_impl(Rng& range_of_tuples)
288  {
289  using tuple_type = ranges::range_value_type_t<Rng>;
290  constexpr auto tuple_size = std::tuple_size<tuple_type>{};
291  constexpr auto indices = std::make_index_sequence<tuple_size>{};
292  std::size_t reserve_size = detail::safe_reserve_size(range_of_tuples);
293 
294  auto tuple_of_ranges = detail::vectorize_tuple<tuple_type>(indices);
295  utility::tuple_for_each(
296  tuple_of_ranges, [reserve_size](auto& rng) { rng.reserve(reserve_size); });
297 
298  for (auto& v : range_of_tuples) {
299  detail::push_unzipped(tuple_of_ranges, std::move(v), indices);
300  }
301 
302  return tuple_of_ranges;
303  }
