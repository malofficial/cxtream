208 template<long NDims, typename Rng, typename ValT = ndim_type_t<Rng, NDims>>
209 Rng& ndim_resize(Rng& vec,
210  const std::vector<std::vector<long>>& vec_size,
211  ValT val = ValT{})
212 {
213  // check that the size is valid
214  assert(vec_size.size() == NDims);
215  static_assert(NDims <= ndims<Rng>{} - ndims<ValT>{});
216  for (std::size_t i = 1; i < vec_size.size(); ++i) {
217  assert(vec_size[i].size() == ranges::accumulate(vec_size[i-1], 0UL));
218  }
219  // build initial indices
220  std::vector<long> vec_size_idx(vec_size.size());
221  // recursively resize
222  detail::ndim_resize_impl<1, NDims>::impl(vec, vec_size, vec_size_idx, val);
223  return vec;
224 }

382  template<long N>
383  struct reshaped_view_impl_go {
384  static auto impl(const std::shared_ptr<std::vector<long>>& shape_ptr)
385  {
386  return ranges::view::chunk((*shape_ptr)[N-2])
387  | ranges::view::transform([shape_ptr](auto subview) {
388  return std::move(subview) | reshaped_view_impl_go<N-1>::impl(shape_ptr);
389  });
390  }
391  };
392 
393  template<>
394  struct reshaped_view_impl_go<1> {
395  static auto impl(const std::shared_ptr<std::vector<long>>&)
396  {
397  return ranges::view::all;
398  }
399  };
400 
401  template<long N, typename Rng>
402  auto reshaped_view_impl(Rng& vec, std::vector<long> shape)
403  {
404  assert(shape.size() == N);
405  auto flat = flat_view(vec);
406 
407  // if -1 present in the shape list, deduce the dimension
408  auto deduced_pos = ranges::find(shape, -1);
409  if (deduced_pos != shape.end()) {
410  auto flat_size = ranges::distance(flat);
411  auto shape_prod = -ranges::accumulate(shape, 1, std::multiplies<>{});
412  assert(flat_size % shape_prod == 0);
413  *deduced_pos = flat_size / shape_prod;
414  }
415 
416  // check that all the requested dimenstions have positive size
417  assert(ranges::all_of(shape, [](long s) { return s > 0; }));
418  // check that the user requests the same number of elements as there really is
419  assert(ranges::distance(flat) == ranges::accumulate(shape, 1, std::multiplies<>{}));
420  // calculate the cummulative product of the shape list in reverse order
421  shape |= ranges::action::reverse;
422  ranges::partial_sum(shape, shape, std::multiplies<>{});
423  // the recursive chunks will share a single copy of the shape list (performance)
424  auto shape_ptr = std::make_shared<std::vector<long>>(std::move(shape));
425  return std::move(flat) | detail::reshaped_view_impl_go<N>::impl(shape_ptr);
426  }
