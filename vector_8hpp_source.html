369  template<long N>
370  struct reshaped_view_impl_go {
371  static auto impl(const std::shared_ptr<std::vector<long>>& shape_ptr)
372  {
373  return ranges::view::chunk((*shape_ptr)[N-2])
374  | ranges::view::transform([shape_ptr](auto subview) {
375  return std::move(subview) | reshaped_view_impl_go<N-1>::impl(shape_ptr);
376  });
377  }
378  };

381  struct reshaped_view_impl_go<1> {

385  }

388  template<long N, typename Rng>
389  auto reshaped_view_impl(Rng& vec, std::vector<long> shape)
390  {
391  assert(shape.size() == N);
392  auto flat = flat_view(vec);
393 
394  // if -1 present in the shape list, deduce the dimension
395  auto deduced_pos = ranges::find(shape, -1);
396  if (deduced_pos != shape.end()) {
397  auto flat_size = ranges::distance(flat);
398  auto shape_prod = -ranges::accumulate(shape, 1, std::multiplies<>{});
399  assert(flat_size % shape_prod == 0);
400  *deduced_pos = flat_size / shape_prod;
401  }
402 
403  // check that all the requested dimenstions have positive size
404  assert(ranges::all_of(shape, [](long s) { return s > 0; }));
405  // check that the user requests the same number of elements as there really is
406  assert(ranges::distance(flat) == ranges::accumulate(shape, 1, std::multiplies<>{}));
407  // calculate the cummulative product of the shape list in reverse order
408  shape |= ranges::action::reverse;
409  ranges::partial_sum(shape, shape, std::multiplies<>{});
410  // the recursive chunks will share a single copy of the shape list (performance)
411  auto shape_ptr = std::make_shared<std::vector<long>>(std::move(shape));
412  return std::move(flat) | detail::reshaped_view_impl_go<N>::impl(shape_ptr);
413  }
