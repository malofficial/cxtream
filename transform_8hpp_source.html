53  // apply fun to each element in tuple of ranges in the given dimension
54  template<std::size_t Dim, std::size_t NOuts>
55  struct wrap_fun_for_dim
56  {
57  template<typename Fun>
58  static constexpr auto impl(Fun fun)
59  {
60  return [fun = std::move(fun)](auto&& tuple_of_ranges) CXTREAM_MUTABLE_LAMBDA_V {
61  assert(utility::same_size(tuple_of_ranges));
62  auto range_of_tuples =
63  std::experimental::apply(ranges::view::zip,
64  std::forward<decltype(tuple_of_ranges)>(tuple_of_ranges))
65  | ranges::view::transform(wrap_fun_for_dim<Dim-1, NOuts>::impl(fun));
66  return utility::unzip_if<(NOuts > 1)>(std::move(range_of_tuples));
67  };
68  }

124  // This function accepts a tuple of references and returns a new tuple
125  // made by moving the values from the original tuple.
126  // If the tuple is of size 1, then it only returns the rvalue
127  // of its element (without wrapping it in a tuple).

150  // wrap the function to be applied only on if the first argument evaluates to true
151  template<std::size_t NOuts>
152  struct wrap_fun_with_cond
153  {
154  template<
155  typename Fun,
156  std::size_t... FromIndices,
157  std::size_t... ToIndices>
158  static constexpr auto impl(
159  Fun fun,
160  std::index_sequence<FromIndices...> from_indices,
161  std::index_sequence<ToIndices...> to_indices)
162  {
163  return [fun = std::move(fun), to_indices]
164  (auto cond, auto&... cols) CXTREAM_MUTABLE_LAMBDA_V {
165  // make a tuple of all arguments, except for the condition
166  auto args_view = std::forward_as_tuple<decltype(cols)...>(cols...);
167  // apply the function if the condition is true
168  if (cond) {
169  // the function is applied only on a subset of the arguments
170  // representing FromColumns
171  return std::experimental::apply(fun,
172  utility::tuple_index_view<FromIndices...>(args_view));
173  // return the original arguments if the condition is false
174  } else {
175  // only a subset of the arguments representing ToColumns is returned
176  // note: We can force std::move in here, because
177  // we are only copying data to themselves.
178  return move_to_maybe_make_tuple<NOuts>::impl(args_view, to_indices);
179  }
180  };
181  }
182  };

270  // wrap the function to be an identity if the dice roll fails
271  template<std::size_t NOuts>
272  struct wrap_fun_with_prob
273  {
274  template<
275  typename Fun,
276  typename Prng,
277  std::size_t... FromIndices,
278  std::size_t... ToIndices>
279  static constexpr auto impl(
280  double prob,
281  Fun fun,
282  Prng& prng,
283  std::index_sequence<FromIndices...> from_indices,
284  std::index_sequence<ToIndices...> to_indices)
285  {
286  return [fun = std::move(fun), &prng, prob, to_indices]
287  (auto&... cols) CXTREAM_MUTABLE_LAMBDA_V {
288  std::uniform_real_distribution<> dis(0, 1);
289  // make a tuple of all arguments
290  auto args_view = std::forward_as_tuple<decltype(cols)...>(cols...);
291  // apply the function if the dice roll succeeds
292  if (prob == 1. || (prob > 0. && dis(prng) < prob)) {
293  // the function is applied only on a subset of the arguments
294  // representing FromColumns
295  return std::experimental::apply(fun,
296  utility::tuple_index_view<FromIndices...>(args_view));
297  // return the original arguments if the dice roll fails
298  } else {
299  // only a subset of the arguments representing ToColumns is returned
300  // note: We can force std::move in here, because
301  // we are only copying data to themselves.
302  return move_to_maybe_make_tuple<NOuts>::impl(args_view, to_indices);
303  }
