65  template<typename Fun, typename... Args>
66  std::future<std::result_of_t<Fun(Args...)>> enqueue(Fun fun, Args... args)
67  {
68  using Ret = std::result_of_t<Fun(Args...)>;
69  std::packaged_task<Ret(Args...)> task{fun};
70  std::future<Ret> future = task.get_future();
71  // Packaged task is non-copyable, so ASIO's post() method cannot handle it.
72  // So we build a shared_ptr of the task and post a lambda
73  // dereferencing and running the task stored in the pointer.
74  auto shared_task = std::make_shared<std::packaged_task<Ret(Args...)>>(std::move(task));
75  auto shared_args = std::make_shared<std::tuple<Args...>>(std::move(args)...);
76  auto asio_task = [task = std::move(shared_task), args = std::move(shared_args)]() {
77  return std::experimental::apply(std::move(*task), std::move(*args));
78  };
79  service_.post(std::move(asio_task));
80  return future;
81  }
