667 template<typename DataTable>
668 std::ostream& operator<<(std::ostream& out, const dataframe<DataTable>& df)
669 {
670  namespace view = ranges::view;
671  // calculate the width of the columns using their longest field
672  std::vector<std::size_t> col_widths = df.raw_cols()
673  | view::transform([](auto&& col) {
674  std::vector<std::size_t> elem_sizes = col
675  | view::transform([](auto& field) { return ranges::size(field); });
676  return ranges::max(elem_sizes) + 2;
677  });
678 
679  auto header = df.header();
680  if (header.size()) {
681  // update col_widths using header widths
682  col_widths = view::zip(col_widths, header)
683  | view::transform([](auto&& tpl) {
684  return std::max(std::get<0>(tpl), std::get<1>(tpl).size() + 2);
685  });
686 
687  // print header
688  for (std::size_t j = 0; j < header.size(); ++j) {
689  out << std::setw(col_widths[j]) << header[j];
690  if (j + 1 < header.size()) out << '|';
691  else out << '\n';
692  }
693 
694  // print header and data separator
695  for (std::size_t j = 0; j < header.size(); ++j) {
696  out << std::setw(col_widths[j]) << std::setfill('-');
697  if (j + 1 < header.size()) out << '-' << '+';
698  else out << '-' << '\n';
699  }
700  out << std::setfill(' ');
701  }
702 
703  // print data
704  for (std::size_t i = 0; i < df.n_rows(); ++i) {
705  for (std::size_t j = 0; j < df.n_cols(); ++j) {
706  out << std::setw(col_widths[j]) << df.raw_rows()[i][j];
707  if (j + 1 < df.n_cols()) out << '|';
708  else out << '\n';
709  }
710  }
711 
712  return out;
713 }
