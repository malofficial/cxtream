23  // Create a random_fill transformation function that can be sent to stream::transform.
24  template<typename FromColumn, typename ToColumn, typename Prng, typename Dist>
25  struct wrap_random_fill_fun_for_transform {
26  long rnddims;
27  Dist dist;
28  std::reference_wrapper<Prng> prng;
29 
30  typename ToColumn::batch_type operator()(typename FromColumn::batch_type& source)
31  {
32  using SourceVector = typename FromColumn::batch_type;
33  using TargetVector = typename ToColumn::batch_type;
34  constexpr long SourceDims = utility::ndims<SourceVector>::value;
35  constexpr long TargetDims = utility::ndims<TargetVector>::value;
36  static_assert(TargetDims <= SourceDims, "stream::random_fill requires"
37  " the target column to have at most the same number of dimensions"
38  " as the source column.");
39  // get the size of the source up to the dimension of the target
40  std::vector<std::vector<long>> target_size = utility::ndim_size<TargetDims>(source);
41  // create, resize, and fill the target with random values
42  TargetVector target;
43  utility::ndim_resize(target, target_size);
44  // make a copy of the stored distribution to avoid races when buffering
45  utility::random_fill(target, rnddims, Dist{dist}, prng.get());
46  return target;
47  }
48  };
